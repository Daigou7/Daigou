// --- LÓGICA DE DATA E HORA (VERSÃO KANJI CORRIGIDA) --- function updateClock() { const now = new Date(); // Hora (HH:MM:SS) const timeString = now.toLocaleTimeString('en-US', { hour12: false }); document.getElementById('clock').innerText = timeString; // Data com Kanjis (Ano年 Mês月 Dia日) const year = String(now.getFullYear()).slice(-2); // Pega os últimos 2 dígitos (ex: 26) const month = String(now.getMonth() + 1).padStart(2, '0'); const day = String(now.getDate()).padStart(2, '0'); document.getElementById('date').innerText = `${year}年${month}月${day}日`; } setInterval(updateClock, 1000); updateClock();
